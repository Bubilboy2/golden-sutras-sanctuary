                messageBox.classList.add('opacity-100', 'text-green-400');

                    messageBox.classList.remove('opacity-100');
                    messageBox.classList.add('opacity-0');
                }, 4000);

                messageBox.textContent = 'Please enter a valid sacred email address.';
                messageBox.classList.remove('opacity-0', 'text-green-400');
                messageBox.classList.add('opacity-100', 'text-red-400');
